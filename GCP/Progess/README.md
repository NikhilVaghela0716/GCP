```bash
javascript:(function () {
    const leaderboard = document.querySelector('.js-lab-leaderboard');
    const gameLabs = document.querySelector('.games-labs');

    if (leaderboard) {
        leaderboard.remove();
    }

    if (gameLabs) {
        gameLabs.classList.remove('games-labs');
        gameLabs.classList.add(
            'lab-show',
            'l-full',
            'no-nav',
            'application-new'
        );
    }
})();
```
